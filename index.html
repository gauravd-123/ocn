Exp 04: Crtical Angle

clc;
clear;
close;
theta_a_degrees=input("enter the value acceptance angle in degrees:- ") //12
theta_a_radians=(theta_a_degrees)*(%pi/180);
delta=input("enter the value of relative refractive index difference:- ") //0.03
NA=sin(theta_a_radians);
disp ("numerical aperture =",NA)
n1=(NA)/(sqrt(2*delta));
disp ("n1=", n1);
n2=n1-(delta*n1);
disp("n2=", n2);
fiy_c=asin(n2/n1);
fiy_c_degrees=(fiy_c)*(180/%pi);
disp('critical angle=',fiy_c_degrees,'degrees');

Exp 07: Skew rays

clc;
clear;
close;
NA= input('Enter numerical aperture= '); //0.12
r=input('Enter angle of skew rays= '); //100
gamma = r/2;
theta_a = asin(NA);
theta_a_deg = theta_a * 180/%pi;
theta_as = asin(NA/cos(gamma));
theta_as_deg = theta_as * 180/%pi;
disp ("Acceptance angle for meridional rays in degree = ",theta_a_deg);
disp ("Acceptance angle for skew rays in degree = ",theta_as_deg);
